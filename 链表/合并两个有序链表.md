# 合并两个有序链表
~~~ c
/**
 * Definition for singly-linked list.
 * struct ListNode 
 {
 *     int val;
 *     struct ListNode *next;
 * };
 */
 struct ListNode *mergeTwoLists(struct ListNode *l1, struct ListNode *l2) 
 {
    struct ListNode head;
    struct ListNode *pre = &head;

    while (l1 && l2) 
    {
        if (l1->val <= l2->val) 
        {
            pre->next = l1;
            l1 = l1->next;
        } else {
            pre->next = l2;
            l2 = l2->next;
        }
        pre = pre->next;
    }

    pre->next = (NULL == l1 ? l2 : l1);
    return head.next;
}
~~~
